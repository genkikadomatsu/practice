# LC 155: Min Stack

class MinStack:

    def __init__(self):
        self.minStack = []

    def push(self, val: int) -> None:
        if len(self.minStack) > 0:
            self.minStack.append((val, min(self.getMin(), val)))
        else:
            self.minStack.append((val, val))

    def pop(self) -> None:
        return self.minStack.pop()[0]

    def top(self) -> int:
        return self.minStack[-1][0]

    def getMin(self) -> int:
        return self.minStack[-1][1]
