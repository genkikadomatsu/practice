# LeetCode 3 Longest substring without repeating characters


class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        start, end = 0, 0
        seen = {}
        maxlen = 0

        for i in range(len(s)):

            if s[i] in seen:
                start = max(start, seen[s[i]] + 1)

            seen[s[i]] = i
            maxlen = max(maxlen, i - start + 1)

        return maxlen
