class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:



        result = []

        def combinations(i, current, sum):

            if sum == target:
                result.append(current)
                return
            
            if sum > target or i >= target:
                return
            
            combinations(i, current + [candidates[i]], sum + candidates[i])
            combinations(i + 1, current, sum)

            return
        
        combinations(0, [], 0)

        return result
