# LeetCode 48
class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """


        n = len(matrix)
        start, end = 0, n

        while start < end:
            for j in range(start, end - 1):

                # overwrite end column with start row
                temp_a = matrix[j][end - 1]
                matrix[j][end - 1] = matrix[start][j]

                # overwrite end row with end column
                temp_b = matrix[end - 1][-j - 1]
                matrix[end - 1][-j - 1] = temp_a

                # overwrite start column with end row
                temp_c = matrix[-j - 1][start]
                matrix[-j - 1][start] = temp_b 

                # overwrite start row with start column
                matrix[start][j] = temp_c 
            
            start += 1
            end -= 1
